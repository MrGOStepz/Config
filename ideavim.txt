"" Source your .vimrc
"source ~/.vimrc

"" -- Suggested options --
" Show a few lines of context around the cursor. Note that this makes the
" text scroll if you mouse-click near the start or end of the window.
set scrolloff=5
set number relativenumber
set idearefactormode=keep
set ideajoin
set surround
set easymotion
" Do incremental searching.
set incsearch
set mouse=a
set clipboard-=unnamed
set clipboard-=autoselect
set clipboard+=ideaput
" Don't use Ex mode, use Q for formatting.

" In insert or command mode, move normally by using Ctrl
inoremap <C-h> <Left>
inoremap <C-j> <Down>
inoremap <C-k> <Up>
inoremap <C-l> <Right>

nnoremap <C-A-j> :action Back<CR>
nnoremap <C-b> :action GotoDeclaration<CR>
nnoremap <C-A-b> :action GotoImplementation<CR>
nnoremap <C-w> :action EditorSelectWord<CR>
nnoremap <C-e> :action EditorEscape<CR>

" Debugging mappings
nnoremap <leader>d :action Debug<CR>
nnoremap <leader>b :action ToggleLineBreakpoint<CR>
nnoremap <F5> :action Resume<CR>
nnoremap <F6> :action StepOver<CR>
nnoremap <F7> :action StepInto<CR>
nnoremap <F8> :action StepOut<CR>
nnoremap <leader>e :action EvaluateExpression<CR>


set NERDTree
let g:NERDTreeMapActivateNode='l'
let g:NERDTreeMapJumpParent='h'

map Q gq
imap <C+[> <ESC>
nnoremp \r :action IdeaVim.ReloadVimRc.reload<CR>
